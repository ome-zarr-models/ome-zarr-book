# Further parallel processing examples

+++

In this chapter, we'll look at some more complex examples of parallel processing Zarr arrays.

## Downsampling an image

If we're downsampling by a factor of two, then our output array will have half the shape of the input array. In 3D, eight input chunks will map to one output chunk.

```{mermaid}
    flowchart LR
        input_1 --> output_i
        input_2 --> output_i
        input_3 --> output_i
        input_4 --> output_i
        input_5 --> output_i
        input_6 --> output_i
        input_7 --> output_i
        input_8 --> output_i

        output_i["Chunk i, j, k"]

        input_1["Chunk 2i, 2j, 2k"]
        input_2["Chunk 2i, 2j, 2k+1"]
        input_3["Chunk 2i, 2j+1, 2k"]
        input_4["Chunk 2i, 2j+1, 2k+1"]
        input_5["Chunk 2i+1, 2j, 2k"]
        input_6["Chunk 2i+1, 2j, 2k+1"]
        input_7["Chunk 2i+1, 2j+1, 2k"]
        input_8["Chunk 2i+1, 2j+1, 2k+1"]
```

Let's call each input region of 8 chunks (2 x 2 x 2) a 'block':

```{code-cell} ipython3
block_size = (2, 2, 2)
```

Then, let's make a function that can take an input array and create an empty output array of the correct size. For our example, the output will be 2x smaller than the input.

```{code-cell} ipython3
import math

def create_output_array(input_array: zarr.Array, block_size: tuple[int, ...]) -> zarr.Array:
    """Make an empty output array of size input_array.shape / block_size"""
    
    input_shape = input_array.shape
    output_array = zarr.zeros(
        [math.ceil(input_shape[i] / block_size[i]) for i, shape in enumerate(input_shape)],
        chunks=input_array.chunks,
        zarr_format=2
    )

    return output_array
```

Let's check it works for our input heart dataset.

```{code-cell} ipython3
from data_helpers import plot_slice
import matplotlib.pyplot as plt

output_array = create_output_array(heart_image, block_size)
print("input image shape", heart_image.shape)
print("output image shape", output_array.shape)

fig, axs = plt.subplots(ncols=2)
plot_slice(heart_image, z_idx=65, ax=axs[0])
plot_slice(output_array, z_idx=65, ax=axs[1])
```

Then, we define a function that can create a mapping from input slices to output slices. This should map each block in the input array to one chunk in the output array.

```{code-cell} ipython3
import itertools

def all_block_indices(
    array: zarr.Array, 
    block_size: tuple[int, ...], 
) -> dict[tuple[slice, ...], tuple[slice, ...]]:
    """
    Generate a dictionary that maps the location of input blocks to output chunks.
    """

    input_to_output_slice = {}
    ndim = len(array.shape)

    input_indices = [range(0, array.shape[i], array.chunks[i] * block_size[i]) for i in range(ndim)]
    chunk_corners = itertools.product(*input_indices)

    input_slices = [
        tuple(
            slice(corner[i], min(corner[i] + array.chunks[i] * block_size[i], array.shape[i]))
            for i in range(ndim)
        )
        for corner in chunk_corners
    ]

    output_slices = []
    for input_slice in input_slices:
        output_slice = tuple(
            slice(
                math.ceil(input_slice[i].start / block_size[i]), 
                math.ceil(input_slice[i].stop / block_size[i])
            )
            for i in range(ndim)
        )
        output_slices.append(output_slice)

    return dict(zip(input_slices, output_slices))
```

```{code-cell} ipython3
all_block_indices(heart_image, block_size)
```

Then let's create a function that can process a single input block:

```{code-cell} ipython3
import skimage
from collections.abc import Callable
import numpy.typing as npt
from typing import Any
import joblib

def apply_to_block(
    f: Callable[[npt.NDArray[Any]], npt.NDArray[Any]],
    input_array: zarr.Array,
    output_array: zarr.Array,
    block_size: tuple[int, ...],
    input_slice: tuple[slice, ...],
    output_slice: tuple[slice, ...]
) -> None:
    """
    Apply a function to input_array[input_slice] and write to output_array[output_slice]

    Parameters
    ----------
    f :
        Function to apply to slice of data.
    input_array :
        Array to read from.
    output_array :
        Array to write to.
    block_size : size of input blocks
    input_slice : slice of input
    output_slice : slice of output
    """
    print(f"Reading index {input_slice}...")
    input_block = input_array[input_slice]

    # Pad to a multiple of the block size
    pads = np.array(input_block.shape) % block_size
    pad_width = [(0, p) for p in pads]
    input_block = np.pad(input_block, pad_width, mode="edge")
    
    processed_block = skimage.measure.block_reduce(
        input_block, block_size=block_size, func=f
    ).astype(input_block.dtype)

    print(f"Writing processed block of size {processed_block.shape} to index {output_slice}...")
    output_array[output_slice] = processed_block


delayed_apply_to_block = joblib.delayed(apply_to_block)
```

Now let's combine this all together to make a function that takes an input array / block size and:
- creates an output array of the right size
- makes joblib jobs for applying a function to every input block

```{code-cell} ipython3
def blockwise_jobs(
    f: Callable[[npt.NDArray[Any]], npt.NDArray[Any]],
    *,
    input_array: zarr.Array,
    block_size: tuple[int, ...],
) -> tuple[zarr.Array, list[joblib.delayed]]:
    """
    Apply a function to all blocks of a Zarr array.

    This returns (an output array of the correct shape, the joblib jobs for each block)

    Parameters
    ----------
    f :
        Function to apply to each block.
    input_array :
        Array to read from.
    block_size:
        The size of the block in chunks.
    """

    output_array = create_output_array(input_array, block_size)
    input_to_output_indices = all_block_indices(input_array, block_size)
        
    return output_array, [
        delayed_apply_to_block(f, input_array, output_array, block_size, input_slice, output_slice)
        for input_slice, output_slice in input_to_output_indices.items()
    ]
```

Then we setup our jobs using `np.mean` as our downsampling function:

```{code-cell} ipython3
import numpy as np
from rich import print as pprint

downsampled_image, jobs = blockwise_jobs(np.mean, input_array=heart_image, block_size=block_size)
print(f"Number of jobs: {len(jobs)}")
print("First job:")
pprint(jobs[0])
```

Then run them all...

```{code-cell} ipython3
executor = joblib.Parallel(n_jobs=2, backend='threading')
executor(jobs);
```

Then check the results - to see our final downsampled image:

```{code-cell} ipython3
fig, axs = plt.subplots(ncols=2)

slice_no = 66
plot_slice(heart_image, z_idx=slice_no, ax=axs[0])
plot_slice(downsampled_image, z_idx=math.ceil(slice_no / block_size[0]), ax=axs[1])
```

## Downsampling a segmentation

+++

Let's make a simple label image we can use as an example, by thresholding the heart:

```{code-cell} ipython3
from skimage.measure import label

heart_image_np = load_heart_data(array_type='numpy')
heart_labels_np = np.zeros_like(heart_image_np)
heart_labels_np[heart_image_np > 50] = 255

heart_labels = zarr.array(heart_labels_np, chunks=(32,32,32))

print("unique label values", np.unique(heart_labels_np))

fig, axs = plt.subplots(ncols=2)
plot_slice(heart_image, z_idx=80, ax=axs[0])
plot_slice(heart_labels, z_idx=80, ax=axs[1])
```

Then we can try using `np.mean` same as before:

```{code-cell} ipython3
downsampled_labels, jobs = blockwise_jobs(np.mean, input_array=heart_labels, block_size=block_size)
executor(jobs);
```

Unfortunately, the result images aren't what we wanted. Our labels of 0 (background) and 255 (foreground) now show fringes of various intermediate pixel values.

```{code-cell} ipython3
fig, axs = plt.subplots(ncols=2)

print("unique label values", np.unique(downsampled_labels[:]))

slice_no = 80
plot_slice(heart_labels, z_idx=slice_no, ax=axs[0])
plot_slice(downsampled_labels, z_idx=math.ceil(slice_no / block_size[0]), ax=axs[1])
```

We can fix this by using a different downsampling function that won't produce intermediate values. For example, `mode`

```{code-cell} ipython3
from scipy.stats import mode

def mode_of_array(array: npt.NDArray[Any], axis: int | None) -> npt.NDArray:
    return mode(array, axis=axis).mode
```

```{code-cell} ipython3
from scipy.stats import mode

downsampled_labels, jobs = blockwise_jobs(mode_of_array, input_array=heart_labels, block_size=block_size)
executor(jobs);
```

```{code-cell} ipython3
fig, axs = plt.subplots(ncols=2)

print("unique label values", np.unique(downsampled_labels[:]))

slice_no = 80
plot_slice(heart_labels, z_idx=slice_no, ax=axs[0])
plot_slice(downsampled_labels, z_idx=math.ceil(slice_no / block_size[0]), ax=axs[1])
```
